print("Hello, World! I'm Syed Atiq and this is my Python journey.")

# в”Җв”Җ Loops вҖ” My Favourite Topic в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
skills = ["Python", "HTML", "CSS", "JavaScript", "Data Science"]
print("\nрҹҡҖ Skills I'm building:")
for i, skill in enumerate(skills, 1):
    print(f"   {i}. {skill}")

# в”Җв”Җ Functions вҖ” Clean & Reusable в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

primes = [x for x in range(2, 50) if is_prime(x)]
print(f"\nвң… Primes below 50: {primes}")
